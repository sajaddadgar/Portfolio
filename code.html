Dynamic Access Control in IoT: Automated Policy Generation with Machine Learning¶

Objectives of the project:

  1. Generate synthetic access control policies using Generative Adversarial Networks (GANs).
  2. Implement a real-time GAN to be able to learn from streams in real-time, instead of a fixed dataset in order to align the characteristics of IoT.
  3. Develop various ML and DL models, from traditional ML techniques to DL techniques to predict access control policies effectively.

Model training¶

Train the GAN:

  1. Generate synthetic data
  2. Train the discriminator model using real and synthetic data
  3. Calculate the loss
  4. Train the generator model using the input and the calculated loss
In [ ]:
X_train_reshaped = np.reshape(X, (X.shape[0], X.shape[1], 1))
epochs = 1000
batch_size = 64

discriminate_loss = []
discriminate_acc = []
generate_loss = []
generate_acc = []

for epoch in range(epochs):

    idx = np.random.randint(0, X_train_reshaped.shape[0], batch_size)
    real_data = X_train_reshaped[idx]

    fake_data = generator.predict(real_data)
    labels_real = np.ones((batch_size, 1))
    labels_fake = np.zeros((batch_size, 1))

    d_loss_real = discriminator.train_on_batch(real_data, labels_real)
    d_loss_fake = discriminator.train_on_batch(fake_data, labels_fake)
    d_loss = 0.5 * np.add(d_loss_real, d_loss_fake)

    noise = np.random.normal(0, 1, (batch_size, X_train_reshaped.shape[1], 1))
    labels_gan = np.ones((batch_size, 1))
    g_loss = gan.train_on_batch(noise, labels_gan)

    discriminate_loss.append(d_loss[0])
    discriminate_acc.append(100 * d_loss[1])
    generate_loss.append(g_loss[0])
    generate_acc.append(100 * g_loss[1])

def real_time_data_generator2(file_path):
    last_modification_time = 0
    while True:
        # Check if the file has been modified
        current_modification_time = os.path.getmtime(file_path)
        if current_modification_time > last_modification_time:
            with open(file_path, 'r') as file:
                lines = file.readlines()
            data = np.array([list(map(float, line.strip().split(','))) for line in lines])
            data = data.reshape((data.shape[0], X.shape[1], 1))
            last_modification_time = current_modification_time
            yield data
        else:
            time.sleep(1)
        if keyboard.is_pressed('q'):
            print("User pressed 'q'")
            break

def evaluate(y_test, prediction):

    accuracy = accuracy_score(y_test, prediction)
    precision = precision_score(y_test, prediction)
    recall = recall_score(y_test, prediction)

    print('Accuracy score: {}'.format(accuracy))
    print('Precision score: {}'.format(precision))
    print('Recall score: {}'.format(recall))

    return accuracy

def confusion_matrix_plot(matrix):
    group_counts = ['{0:0.0f}'.format(value) for value in matrix.flatten()]
    group_percentages = ['{0:.2%}'.format(value) for value in matrix.flatten() / np.sum(matrix)]
    labels = [f'{a}\n{b}' for a, b in zip(group_counts, group_percentages)]
    labels = np.asarray(labels).reshape(2, 2)
    sns.heatmap(matrix, annot=labels, fmt='', cmap=sns.light_palette("seagreen", as_cmap=True))

def create_model(dropout_rate):
  model = Sequential()
  model.add(Dense(units = 64, input_shape=(X.shape[1],), activation = 'relu'))
  model.add(Dropout(dropout_rate))
  model.add(Dense(units = 32, activation = 'relu'))
